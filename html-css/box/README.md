	/* border-radius */
	*{border-radius:10px}
	div{border-top-right-radius:10px}

	/* border-width */
	*{border-width:10px}
	div{border-top-width:10px}

	/* border-style */
	*{border-style:solid}
	div{border-right-style:solid}

	/* border-color */
	*{border-color:red}
	div{border-bottom-color:red}
